   28: =head1 DESCRIPTION
   29: 
   30: Exports a single function, C<get_params>, which returns a given value.
   31: 
   32: When used hand-in-hand with L<Params::Validate::Strict> and L<Return::Set>,
   33: you should be able to formally specify the input and output sets for a method.
   34: 
   35: =head1 SYNOPSIS
   36: 
   37:     use Params::Get;
   38:     use Params::Validate::Strict;
   39: 
   40:     sub where_am_i
   41:     {
   42:         my $params = Params::Validate::Strict::validate_strict({
   43:             args => Params::Get::get_params(undef, \@_),
   44:             schema => {
   45:                 'latitude' => {
   46:                     type => 'number',
   47:                     min => -90,
   48:                     max => 90
   49:                 }, 'longitude' => {
   50:                     type => 'number',
   51:                     min => -180,
   52:                     max => 180
   53:                 }
   54:             }
   55:         });
   56: 
   57:         print 'You are at ', $params->{'latitude'}, ', ', $params->{'longitude'}, "\n";
   58:     }
   59: 
   60:     where_am_i(latitude => 0.3, longitude => 124);
   61:     where_am_i({ latitude => 3.14, longitude => -155 });
   62: 
   63: =head1	METHODS
   64: 
   65: =head2 get_params
   66: 
   67: Parse the arguments given to a function.
   68: Processes arguments passed to methods and ensures they are in a usable format,
   69: allowing the caller to call the function in any way that they want
   70: e.g. `foo('bar')`, `foo(arg => 'bar')`, `foo({ arg => 'bar' })` all mean the same
   71: when called with
   72: 
   73:     get_params('arg', @_);
   74: 
   75: or
   76: 
   77:     get_params('arg', \@_);
   78: 
   79: Some people like this sort of model, which is also supported.
   80: 
   81:     use MyClass;
   82: 
   83:     my $str = 'hello world';
   84:     my $obj = MyClass->new($str, { type => 'string' });
   85: 
   86:     package MyClass;
   87: 
   88:     use Params::Get;
   89: 
   90:     sub new {
   91:         my $class = shift;
   92:         my $rc = Params::Get::get_params('value', \@_);
   93: 
   94:         return bless $rc, $class;
   95:     }
   96: 
   97: =head2 The C<$default> Parameter
   98: 
   99: The first argument is the C<$default> parameter controls how single-argument calls are interpreted and provides
  100: a default key name for parameter extraction in those cases.
  101: 
  102: When no arguments are provided with a defined C<$default>:
  103: 
  104:     get_params('required'); # Throws usage error
  105: 
  106: The function requires either arguments or an undefined C<$default>.
  107: 
  108: =head3 Usage Examples
  109: 
  110: =over 2
  111: 
  112: =item * Simple scalar parameter:
  113: 
  114:     sub set_country {
  115:         my $params = get_params('country', @_);
  116:         # Accepts: set_country('US')
  117:         # Returns: { country => 'US' }
  118:     }
  119: 
  120: =item * Object constructor with options:
  121: 
  122:     sub new {
  123:         my $class = shift;
  124:         my $params = get_params('value', @_);
  125:         # Accepts: MyClass->new($object)
  126:         # Accepts: MyClass->new($object, { option => 'value' })
  127:         # Returns: { value => $object } or { value => $object, option => 'value' }
  128:     }
  129: 
  130: =item * Hash parameter:
  131: 
  132:     sub configure {
  133:         my $params = get_params('config', @_);
  134:         # Accepts: configure({ db => 'mysql', host => 'localhost' })
  135:         # Returns: { config => { db => 'mysql', host => 'localhost' } }
  136:     }
  137: 
  138: =item * Without default (named parameters only):
  139: 
  140:     sub process {
  141:         my $params = get_params(undef, @_);
  142:         # Accepts: process(name => 'John', age => 30)
  143:         # Returns: { name => 'John', age => 30 }
  144:     }
  145: 
  146: =back
  147: 
  148: =head3 Caveats
  149: 
  150: =over 2
  151: 
  152: =item * When C<$default> is defined and no arguments are provided, an error is thrown
  153: 
  154: =item * There's no way to specify that a default parameter is optional
  155: 
  156: =item * Single hash references always bypass the default parameter naming
  157: 
  158: =back
